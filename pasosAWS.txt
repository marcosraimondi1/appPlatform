1. sudo apt update

-> instalar docker

2. wget https://get.docker.com/    // descargar instalador docker

3. mv index.html install_docker.sh // cambiar nombre archivo

4. chmod 777 install_docker.sh  // otorga al archivo acceso total

5. ./install_docker.sh   // ejecutar instalador

6. sudo curl -L "https://github.com/docker/compose/releases/download/1.29.2/docker-compose-$(uname -s)-$(uname -m)" -o /usr/local/bin/docker-compose

7. sudo chmod +x /usr/local/bin/docker-compose

7. sudo rm install_docker.sh // eliminar instalador

8. sudo apt install git

9. sudo apt install npm

-> para instalar nvm y usar node y npm actualizados

11. sudo curl https://raw.githubusercontent.com/creationix/nvm/master/install.sh | bash

-> reiniciar la consola para que se efectuen los cambios

14. cd webserver

15. sudo rm frontend -> delete frontend unnecesary folder

16. configurar archivo .env
